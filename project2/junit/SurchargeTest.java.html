SurchargeTest.java 
import fleet.*;
import util.Date;
import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

/**
 * Tests the surcharge() method for Sedan, Utility, and Truck classes.
 * @author alexg
 */
public class SurchargeTest {

    @Test
    public void testSedanSurcharge() {
        Sedan sedan = new Sedan("11111S", new Date("1/1/2020"), "Honda", 1000, Campus.BUSCH);
        // No surcharge
        assertEquals(0.0, sedan.surcharge(100, false), 0.01);
        // Below max
        assertEquals(25.0, sedan.surcharge(100, true), 0.01);
        // Above max
        assertEquals(32.99, sedan.surcharge(500, true), 0.01);
    }

    @Test
    public void testUtilitySurcharge() {
        Utility util = new Utility("22222D", new Date("1/1/2020"), "Ford", 2000, Campus.NEWARK);
        // No surcharge
        assertEquals(0.0, util.surcharge(100, false), 0.01);
        // Below max
        assertEquals(30.0, util.surcharge(100, true), 0.01);
        // Above max
        assertEquals(35.99, util.surcharge(200, true), 0.01);
    }

    @Test
    public void testTruckSurcharge() {
        Truck truck = new Truck("33333X", new Date("1/1/2020"), "Chevy", 3000, Campus.CAMDEN);
        // No surcharge
        assertEquals(0.0, truck.surcharge(100, false), 0.01);
        // With surcharge (flat)
        assertEquals(39.99, truck.surcharge(100, true), 0.01);
    }
}
