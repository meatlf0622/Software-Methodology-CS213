import util.Date;
import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

/**
 * Tests the isValid() method in the Date class.
 * (4 invalid and 2 valid cases)
 * @author alexg
 */
public class DateTest {
    
    @Test
    public void testInvalidDates() {
        assertFalse(new Date("0/1/2000").isValid(), "Month 0 invalid");
        assertFalse(new Date("13/11/2000").isValid(), "Month 13 invalid");
        assertFalse(new Date("2/29/1999").isValid(), "1999 not leap year");
        assertFalse(new Date("6/31/2000").isValid(), "June has 30 days");
    }

    @Test
    public void testValidDates() {
        assertTrue(new Date("2/29/2020").isValid(), "Leap year date valid");
        assertTrue(new Date("12/31/2025").isValid(), "Normal date valid");
    }
}
