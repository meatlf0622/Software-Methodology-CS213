VehicleTest.java 
import fleet.*;
import util.Date;
import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

/**
 * Tests compareTo() method of Vehicle and subclasses.
 * @author alexg
 */
public class VehicleTest {

    @Test
    public void testSedanCompareTo() {
        Sedan s1 = new Sedan("11111S", new Date("1/1/2020"), "Toyota", 1000, Campus.BUSCH);
        Sedan s2 = new Sedan("22222S", new Date("1/1/2020"), "Toyota", 1000, Campus.BUSCH);
        Sedan s3 = new Sedan("11111S", new Date("1/1/2020"), "Toyota", 1000, Campus.BUSCH);

        assertTrue(s1.compareTo(s2) < 0);
        assertTrue(s2.compareTo(s1) > 0);
        assertEquals(0, s1.compareTo(s3));
    }

    @Test
    public void testUtilityCompareTo() {
        Utility u1 = new Utility("33333D", new Date("1/1/2020"), "Ford", 2000, Campus.COOK);
        Utility u2 = new Utility("44444D", new Date("1/1/2020"), "Ford", 2000, Campus.COOK);
        Utility u3 = new Utility("33333D", new Date("1/1/2020"), "Ford", 2000, Campus.COOK);

        assertTrue(u1.compareTo(u2) < 0);
        assertTrue(u2.compareTo(u1) > 0);
        assertEquals(0, u1.compareTo(u3));
    }

    @Test
    public void testTruckCompareTo() {
        Truck t1 = new Truck("55555X", new Date("1/1/2020"), "Chevy", 3000, Campus.NEWARK);
        Truck t2 = new Truck("66666X", new Date("1/1/2020"), "Chevy", 3000, Campus.NEWARK);
        Truck t3 = new Truck("55555X", new Date("1/1/2020"), "Chevy", 3000, Campus.NEWARK);

        assertTrue(t1.compareTo(t2) < 0);
        assertTrue(t2.compareTo(t1) > 0);
        assertEquals(0, t1.compareTo(t3));
    }
}
