package fleet;

import util.List;
import util.Sort;
import java.text.DecimalFormat;
import java.util.Comparator;

/**
 * Maintains a list of completed trips.
 * @author alexg
 */
public class TripList extends List<Trip> {

    public void sortByLicenseAndDate() {
        Sort.sort(this, Comparator
                .comparing((Trip t) -> t.getVehicle().getLicense())
                .thenComparing(t -> t.toString())); // fallback textual
    }

    /**
     * Print cost report grouped by department.
     */
    public void printCharges() {
        if (isEmpty()) {
            System.out.println("There is no archived trips for the cost report.");
            return;
        }

        DecimalFormat money = new DecimalFormat("$#,##0.00");
        System.out.println("*List of charges ordered by department.");

        // In a full system, youâ€™d group by department; here, simple list
        for (Trip t : this) {
            double charge = t.getVehicle().charge(t.getMileageUsed());
            double sur = t.getVehicle().surcharge(t.getMileageUsed(), t.hasSurcharge());
            double total = charge + sur;
            String surText = (sur == 0.0) ? "no" : money.format(sur);
            System.out.printf("\t%s%n\t\t[charge: %s] [surcharge: %s] [total charge: %s]%n",
                    t, money.format(charge), surText, money.format(total));
        }
        System.out.println("*end of list.");
    }
}
