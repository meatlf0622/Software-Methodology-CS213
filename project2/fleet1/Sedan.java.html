package fleet;

/**
 * $1.79 per mile, surcharge $0.25 per mile up to $32.99.
 */
public class Sedan extends Vehicle {
    private static final double RATE = 1.79;
    private static final double SURCHARGE_RATE = 0.25;
    private static final double SURCHARGE_MAX = 32.99;

    public Sedan(String license, util.Date date, String make, int mileage, Campus campus) {
        super(license, date, make, mileage, campus);
    }

    @Override
    public double charge(int mileageUsed) {
        return mileageUsed * RATE;
    }

    @Override
    public double surcharge(int mileageUsed, boolean surcharge) {
        if (!surcharge) return 0.0;
        double cost = mileageUsed * SURCHARGE_RATE;
        return Math.min(cost, SURCHARGE_MAX);
    }
}
