package fleet;

/**
 * $2.99 per mile, flat surcharge of $39.99.
 */
public class Truck extends Vehicle {
    private static final double RATE = 2.99;
    private static final double SURCHARGE_FLAT = 39.99;

    public Truck(String license, util.Date date, String make, int mileage, Campus campus) {
        super(license, date, make, mileage, campus);
    }

    @Override
    public double charge(int mileageUsed) {
        return mileageUsed * RATE;
    }

    @Override
    public double surcharge(int mileageUsed, boolean surcharge) {
        return surcharge ? SURCHARGE_FLAT : 0.0;
    }
}
