package fleet;

import util.Date;

/**
 * Abstract representation of a vehicle in the fleet.
 * @author alexg
 */
public abstract class Vehicle implements Comparable<Vehicle> {
    protected String license;      // unique 6-character plate
    protected String make;
    protected Date date;
    protected int mileage;
    protected Campus campus;       // current location

    public Vehicle(String license, Date date, String make, int mileage, Campus campus) {
        this.license = license;
        this.date = date;
        this.make = make;
        this.mileage = mileage;
        this.campus = campus;
    }

    // Charge per mile used
    public abstract double charge(int mileageUsed);

    public abstract double surcharge(int mileageUsed, boolean surcharge);

    public String getLicense() { return license; }
    public String getMake() { return make; }
    public Date getDate() { return date; }
    public int getMileage() { return mileage; }
    public Campus getCampus() { return campus; }
    public void setCampus(Campus c) { campus = c; }
    public void setMileage(int m) { mileage = m; }

    @Override
    public int compareTo(Vehicle other) {
        return this.license.compareTo(other.license);
    }

    @Override
    public boolean equals(Object obj) {
        if (!(obj instanceof Vehicle)) return false;
        return license.equals(((Vehicle)obj).license);
    }

    @Override
    public String toString() {
        String type = this.getClass().getSimpleName().toLowerCase();
        return String.format("%s[%s:%s] %s [mileage:%d] [%s:%s]",
                license, make.toUpperCase(), type, date.toString(),
                mileage, campus.name().substring(0,1) + campus.name().substring(1).toLowerCase(),
                campus.getCity());
    }
}
