package fleet;

/**
 * Enum representing all campuses and their corresponding cities.
 * Each campus has a city name associated with it.
 * @author alexg
 */
public enum Campus {
    BUSCH("New Brunswick"),
    LIVINGSTON("New Brunswick"),
    COOK("New Brunswick"),
    NEWARK("Newark"),
    CAMDEN("Camden");

    private final String city;

    Campus(String city) { this.city = city; }

    public String getCity() { return city; }

    @Override
    public String toString() { return name().substring(0,1) + name().substring(1).toLowerCase(); }

    public static Campus fromString(String s) {
        if (s == null) return null;
        for (Campus c : values()) {
            if (c.name().equalsIgnoreCase(s)) return c;
        }
        return null;
    }
}
