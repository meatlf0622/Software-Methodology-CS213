package fleet;

import util.List;

/**
 * Holds active bookings and supports add/remove operations.
 * @author alexg
 */
public class Reservation extends List<Booking> {

    /** Finds a booking by license plate and start date. */
    public Booking findBooking(String license, util.Date start) {
        for (Booking b : this) {
            if (b.getVehicle().getLicense().equals(license)
                    && b.getStartDate().equals(start))
                return b;
        }
        return null;
    }

    /** Cancels a booking if it exists. */
    public boolean cancelBooking(String license, util.Date start) {
        Booking b = findBooking(license, start);
        if (b != null) {
            remove(b);
            return true;
        }
        return false;
    }

    /** Prints all active bookings. */
    public void printAll() {
        if (isEmpty()) {
            System.out.println("There is no booking record.");
            return;
        }
        System.out.println("*List of reservations ordered by location/license plate/beginning date.");
        for (Booking b : this)
            System.out.println(b);
        System.out.println("*end of list.");
    }
}
