package fleet;

import util.Date;

/**
 * Represents a booking request made by an employee.
 * @author alexg
 */
public class Booking {
    private Vehicle vehicle;
    private Date startDate;
    private Date endDate;
    private String employee;
    private Campus dropoff; // new in Project 2

    public Booking(Vehicle vehicle, Date startDate, Date endDate, String employee, Campus dropoff) {
        this.vehicle = vehicle;
        this.startDate = startDate;
        this.endDate = endDate;
        this.employee = employee;
        this.dropoff = dropoff;
    }

    public Vehicle getVehicle() { return vehicle; }
    public Date getStartDate() { return startDate; }
    public Date getEndDate() { return endDate; }
    public String getEmployee() { return employee; }
    public Campus getDropoff() { return dropoff; }

    @Override
    public String toString() {
        return String.format("%s:%s [%s:%s] %s ~ %s [drop off:%s] [%s]",
                vehicle.getLicense(), vehicle.getMake(),
                vehicle.getCampus(), vehicle.getCampus().getCity(),
                startDate, endDate, dropoff, employee);
    }
}
