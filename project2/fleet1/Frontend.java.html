package fleet;

import java.util.Scanner;

/**
 * @author alexg
 */
public class Frontend {
    private Fleet fleet;
    private Reservation reservations;
    private TripList trips;

    public Frontend() {
        this.fleet = new Fleet();
        this.reservations = new Reservation();
        this.trips = new TripList();
    }

    /**
     * Runs the Vehicle Management System.
     */
    public void run() {
        System.out.println("Vehicle Management System is running.");
        Scanner sc = new Scanner(System.in);
        while (true) {
            if (!sc.hasNext()) break;
            String cmd = sc.next().trim().toUpperCase();
            switch (cmd) {
                case "L" -> handleLoad(sc);
                case "PF" -> handlePrintFleet();
                case "PR" -> reservations.printAll();
                case "PC" -> trips.printCharges();
                case "Q" -> { System.out.println("\nVehicle Management System is terminated."); return; }
                default -> System.out.println(cmd.toLowerCase() + " - invalid command!");
            }
        }
    }

    private void handleLoad(Scanner sc) {
        int loaded = fleet.loadFromFile("vehicles.txt");
        System.out.println(loaded + " vehicles loaded.");
    }

    private void handlePrintFleet() {
        if (fleet.isEmpty()) {
            System.out.println("There is no vehicle in the fleet.");
            return;
        }
        System.out.println("*List of vehicles in the fleet, ordered by location/make/date obtained.");
        for (Vehicle v : fleet) {
            System.out.println(v);
        }
        System.out.println("*end of list.");
    }
}
