package fleet;

import util.List;
import util.Date;
import java.io.File;
import java.io.FileNotFoundException;
import java.util.Scanner;

/**
 * Manages the collection of vehicles.
 * Extends the generic util.List implementation.
 * @author alexg
 */
public class Fleet extends List<Vehicle> {

    /**
     * Loads vehicles from vehicles.txt and returns count loaded.
     */
    public int load(Scanner scanner) {
        int count = 0;
        while (scanner.hasNextLine()) {
            String line = scanner.nextLine().trim();
            if (line.isEmpty()) continue;

            Scanner s = new Scanner(line);
            try {
                String license = s.next();
                String dateStr = s.next();
                String make = s.next();
                int mileage = s.nextInt();
                String campusStr = s.next();

                Campus campus = Campus.fromString(campusStr);
                if (campus == null) continue;

                // avoid duplicates
                boolean exists = false;
                for (Vehicle v : this) {
                    if (v.getLicense().equals(license)) { exists = true; break; }
                }
                if (exists) continue;

                char typeChar = license.charAt(license.length() - 1);
                Vehicle v = null;
                Date date = new Date(dateStr);

                switch (typeChar) {
                    case 'S': v = new Sedan(license, date, make, mileage, campus); break;
                    case 'D': v = new Utility(license, date, make, mileage, campus); break;
                    case 'X': v = new Truck(license, date, make, mileage, campus); break;
                    default: break;
                }

                if (v != null) {
                    this.add(v);
                    count++;
                }

            } catch (Exception ignore) {
                // skip malformed lines
            } finally {
                s.close();
            }
        }
        return count;
    }

    public int loadFromFile(String filePath) {
        try (Scanner fileScanner = new Scanner(new File(filePath))) {
            return load(fileScanner);
        } catch (FileNotFoundException e) {
            System.out.println("File not found: " + filePath);
            return 0;
        }
    }
}
