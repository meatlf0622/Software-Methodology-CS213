package fleet;

import util.Date;

/**
 * @author alexg
 */
public class Trip {
    private Vehicle vehicle;
    private Date startDate;
    private Date endDate;
    private int oldMileage;
    private int newMileage;
    private int mileageUsed;
    private boolean surcharge; // new flag for Project 2
    private Campus pickup;
    private Campus dropoff;

    public Trip(Vehicle vehicle, Date start, Date end,
                int oldMileage, int newMileage, Campus pickup, Campus dropoff) {
        this.vehicle = vehicle;
        this.startDate = start;
        this.endDate = end;
        this.oldMileage = oldMileage;
        this.newMileage = newMileage;
        this.mileageUsed = newMileage - oldMileage;
        this.pickup = pickup;
        this.dropoff = dropoff;
        this.surcharge = (pickup != dropoff);
    }

    public Vehicle getVehicle() { return vehicle; }
    public int getMileageUsed() { return mileageUsed; }
    public boolean hasSurcharge() { return surcharge; }
    public Campus getPickup() { return pickup; }
    public Campus getDropoff() { return dropoff; }

    @Override
    public String toString() {
        String mark = surcharge ? "**" : "";
        return String.format(
                "%s %s ~ %s mileage(old): %d mileage(new): %d mileage(used): %d [dropped off: %s%s] [picked up: %s]",
                vehicle.getLicense(), startDate, endDate,
                oldMileage, newMileage, mileageUsed,
                dropoff, mark, pickup);
    }

    public double totalCharge() {
        double base = vehicle.charge(mileageUsed);
        double extra = vehicle.surcharge(mileageUsed, surcharge);
        return base + extra;
    }
}
