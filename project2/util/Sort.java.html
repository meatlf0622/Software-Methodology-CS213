package util;

import java.util.Comparator;

/**
 * Utility class providing static sorting methods for List<E>.
 * Uses a simple insertion sort (small lists).
 * @author alexg
 */
public class Sort {

    // Sorts a List of Comparable objects in ascending order.
    public static <E extends Comparable<E>> void sort(List<E> list) {
        for (int i = 1; i < list.size(); i++) {
            E key = list.get(i);
            int j = i - 1;
            while (j >= 0 && list.get(j).compareTo(key) > 0) {
                list.set(j + 1, list.get(j));
                j--;
            }
            list.set(j + 1, key);
        }
    }

    // Sorts a List using a Comparator
    public static <E> void sort(List<E> list, Comparator<E> comp) {
        for (int i = 1; i < list.size(); i++) {
            E key = list.get(i);
            int j = i - 1;
            while (j >= 0 && comp.compare(list.get(j), key) > 0) {
                list.set(j + 1, list.get(j));
                j--;
            }
            list.set(j + 1, key);
        }
    }
}
